SELECT TOP(20)
    p.dni AS dni_empleado,
    -- Genera un ID_ROL determinista basado en el DNI (entre 1 y 7)
    (ABS(CHECKSUM(CAST(p.dni AS VARCHAR))) % 7) + 1 AS id_rol,
    -- Genera un ID_TURNO determinista basado en el DNI (entre 1 y 5)
    (ABS(CHECKSUM(CAST(p.dni AS VARCHAR))) % 5) + 1 AS id_turno
FROM
    persona p
WHERE
    p.dni NOT IN (SELECT dni_cliente FROM cliente)
ORDER BY
    p.dni; -- Ordenar por DNI para asegurar la misma seleccin
